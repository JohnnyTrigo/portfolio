01 · O desafio

O problema que
esse projeto resolve

Times de operações de telecomunicações gerenciam centenas de demandas mensais — atualizações de tabelas de preços, configurações de produto, relatórios executivos — sem nenhuma visibilidade centralizada de performance.

Os dados existem, mas estão em silos: planilhas, e-mails, Planner, sistemas legados. Sem integração, sem histórico, sem SLA monitorado em tempo real.

01
Dados em silos

Informações espalhadas entre Planner, Forms e planilhas Excel sem integração. Nenhuma fonte central da verdade. A equipe gasta horas compilando relatórios manualmente.

02
SLA invisível

Sem monitoramento de lead time ou aderência a SLA em tempo real. Violações descobertas somente no relatório mensal — tarde demais para ação corretiva.

03
Retrabalho oculto

Taxa de correções não rastreada por categoria ou analista. Impossível identificar padrões de qualidade, gargalos e oportunidades de melhoria de processo.

-- ── Q1: KPI Executivo por Mês ─────────────────────────────────────────────
-- Responde: Volume · Aderência SLA · Retrabalho · Lead Time
SELECT
    d.nr_ano,
    d.nm_mes,
    COUNT(*)                                                AS Total_Solicitacoes,
    SUM(CASE WHEN s.fl_dentro_sla = 1 THEN 1 ELSE 0 END)  AS Dentro_SLA,
    SUM(CASE WHEN s.fl_correcao   = 1 THEN 1 ELSE 0 END)  AS Com_Correcao,
    ROUND(
        100.0 * SUM(CASE WHEN s.fl_dentro_sla = 1 THEN 1 ELSE 0 END)
              / NULLIF(SUM(CASE WHEN s.fl_dentro_sla IS NOT NULL
                               THEN 1 ELSE 0 END), 0),
    2)                                                      AS Pct_Aderencia_SLA,
    ROUND(
        100.0 * SUM(CASE WHEN s.fl_correcao = 1 THEN 1 ELSE 0 END)
              / NULLIF(COUNT(*), 0),
    2)                                                      AS Pct_Retrabalho,
    ROUND(AVG(s.nr_lead_time_horas), 1)                    AS Lead_Time_Medio_h
FROM ops.fct_solicitacoes  s
JOIN ops.dim_data          d  ON s.sk_data_abertura = d.sk_data
GROUP BY d.nr_ano, d.nr_mes, d.nm_mes
ORDER BY d.nr_ano, d.nr_mes;
-- ── Q2: Ranking de Analistas — Score Composto ─────────────────────────────
-- Score = 70% Aderência SLA + 30% Qualidade (inverso do retrabalho)
SELECT
    a.nm_analista,
    a.nm_squad,
    a.nm_nivel,
    COUNT(*)                                                           AS Total,
    ROUND(AVG(s.nr_lead_time_horas), 1)                                AS Lead_Time_h,
    ROUND(
        100.0 * SUM(CASE WHEN s.fl_dentro_sla = 1 THEN 1 ELSE 0 END)
              / NULLIF(SUM(CASE WHEN s.fl_dentro_sla IS NOT NULL
                               THEN 1 ELSE 0 END), 0),
    1)                                                                 AS Pct_SLA,
    ROUND(
        100.0 * SUM(CASE WHEN s.fl_correcao = 1 THEN 1 ELSE 0 END)
              / NULLIF(COUNT(*), 0),
    1)                                                                 AS Pct_Retrabalho,
    -- Score ponderado: 70% SLA · 30% qualidade
    ROUND(
        (100.0 * SUM(CASE WHEN s.fl_dentro_sla = 1 THEN 1 ELSE 0 END)
               / NULLIF(SUM(CASE WHEN s.fl_dentro_sla IS NOT NULL
                               THEN 1 ELSE 0 END), 0)) * 0.7
      + (1 - (100.0 * SUM(CASE WHEN s.fl_correcao = 1 THEN 1 ELSE 0 END)
                         / NULLIF(COUNT(*), 0)) / 100) * 100 * 0.3,
    1)                                                                 AS Score
FROM ops.fct_solicitacoes  s
JOIN ops.dim_analista      a  ON s.sk_analista = a.sk_analista
WHERE s.nm_status = 'Concluída'
GROUP BY a.nm_analista, a.nm_squad, a.nm_nivel
ORDER BY Score DESC;
-- ── Q3: Tendência com Média Móvel 3M + Variação YoY ──────────────────────
-- Suaviza sazonalidade e compara com mesmo período do ano anterior
WITH base AS (
    SELECT
        d.nr_ano, d.nr_mes, d.nm_mes,
        COUNT(*)                                                   AS total,
        SUM(CASE WHEN s.fl_dentro_sla = 1 THEN 1.0 ELSE 0 END)
          / NULLIF(SUM(CASE WHEN s.fl_dentro_sla IS NOT NULL
                           THEN 1.0 ELSE 0 END), 0)                AS pct_sla,
        AVG(s.nr_lead_time_horas)                                  AS lead_time
    FROM ops.fct_solicitacoes s
    JOIN ops.dim_data         d ON s.sk_data_abertura = d.sk_data
    WHERE s.nm_status = 'Concluída'
    GROUP BY d.nr_ano, d.nr_mes, d.nm_mes
)
SELECT
    nr_ano, nm_mes,
    total                                                          AS Volume,
    ROUND(pct_sla * 100, 1)                                        AS Pct_SLA,
    -- Média móvel 3 meses (janela deslizante)
    ROUND(AVG(CAST(total AS FLOAT))
          OVER (ORDER BY nr_ano, nr_mes
                ROWS BETWEEN 2 PRECEDING AND CURRENT ROW), 1)      AS MM3_Volume,
    ROUND(AVG(pct_sla * 100)
          OVER (ORDER BY nr_ano, nr_mes
                ROWS BETWEEN 2 PRECEDING AND CURRENT ROW), 1)      AS MM3_Pct_SLA,
    -- Crescimento YoY via LAG de 12 períodos
    total - LAG(total, 12) OVER (ORDER BY nr_ano, nr_mes)          AS Var_YoY,
    ROUND(100.0
        * (total - LAG(total, 12) OVER (ORDER BY nr_ano, nr_mes))
        / NULLIF(LAG(total, 12) OVER (ORDER BY nr_ano, nr_mes), 0),
    1)                                                             AS Pct_YoY
FROM base
ORDER BY nr_ano, nr_mes;
-- ── Q4: Detecção de Anomalias via Z-Score ─────────────────────────────────
-- Z > 2.0 = Anomalia · Z > 1.5 = Atenção · identifica gargalos ocultos
WITH stats AS (
    SELECT
        sk_categoria,
        AVG(nr_lead_time_horas)    AS media,
        STDEV(nr_lead_time_horas)  AS desvio
    FROM ops.fct_solicitacoes
    WHERE nm_status = 'Concluída'
    GROUP BY sk_categoria
)
SELECT
    s.cd_solicitacao,
    a.nm_analista,
    c.nm_categoria,
    s.nr_lead_time_horas                                           AS Lead_Time_Real,
    ROUND(st.media,  1)                                            AS Media_Categoria,
    ROUND(st.desvio, 1)                                            AS Desvio_Padrao,
    ROUND((s.nr_lead_time_horas - st.media)
        / NULLIF(st.desvio, 0), 2)                                 AS Z_Score,
    CASE
        WHEN (s.nr_lead_time_horas - st.media)
           / NULLIF(st.desvio, 0) > 2.0  THEN '🔴 ANOMALIA'
        WHEN (s.nr_lead_time_horas - st.media)
           / NULLIF(st.desvio, 0) > 1.5  THEN '🟡 ATENÇÃO'
        ELSE '🟢 Normal'
    END                                                            AS Status
FROM ops.fct_solicitacoes  s
JOIN ops.dim_analista      a  ON s.sk_analista  = a.sk_analista
JOIN ops.dim_categoria     c  ON s.sk_categoria = c.sk_categoria
JOIN stats                 st ON s.sk_categoria = st.sk_categoria
WHERE s.nm_status = 'Concluída'
  AND (s.nr_lead_time_horas - st.media)
    / NULLIF(st.desvio, 0) > 1.5
ORDER BY Z_Score DESC;
-- ── Q5: View otimizada para Power BI ─────────────────────────────────────
-- Fonte única da verdade · Suporta DirectQuery e Import Mode
CREATE OR ALTER VIEW ops.vw_solicitacoes_pbi AS
SELECT
    s.cd_solicitacao,
    -- Datas
    da.dt_data       AS dt_abertura,
    da.nr_ano,  da.nr_mes,  da.nm_mes AS nm_mes_abertura,
    da.nm_trimestre,
    df.dt_data       AS dt_fechamento,
    -- Analista
    a.nm_analista,  a.nm_squad,  a.nm_nivel,
    -- Categoria
    c.nm_grupo,  c.nm_categoria,
    c.nr_sla_horas,  c.fl_critico,
    -- Produto
    p.nm_linha,  p.nm_segmento,  p.nm_produto,
    -- Fatos
    s.nm_status,  s.nm_prioridade,  s.nm_canal_entrada,
    s.nr_lead_time_horas,
    s.fl_dentro_sla,  s.fl_correcao,
    -- Colunas calculadas prontas para DAX (evitam medidas repetitivas)
    CASE WHEN s.fl_dentro_sla = 1
         THEN 'Dentro do SLA'   ELSE 'Fora do SLA'    END  AS status_sla,
    CASE WHEN s.fl_correcao = 1
         THEN 'Com Retrabalho'  ELSE 'Sem Retrabalho'  END  AS status_correcao,
    CASE WHEN s.nm_status IN ('Aberta','Em Andamento')
         THEN 1 ELSE 0 END                                   AS fl_aberta,
    ROUND(s.nr_lead_time_horas / 8.0, 1)                     AS lead_time_dias
FROM ops.fct_solicitacoes  s
JOIN ops.dim_data          da ON s.sk_data_abertura   = da.sk_data
LEFT JOIN ops.dim_data     df ON s.sk_data_fechamento = df.sk_data
JOIN ops.dim_analista       a ON s.sk_analista         = a.sk_analista
JOIN ops.dim_categoria      c ON s.sk_categoria        = c.sk_categoria
JOIN ops.dim_produto        p ON s.sk_produto          = p.sk_produto;
04 · Business Intelligence

DAX que responde
perguntas de negócio

Cada medida foi projetada para responder uma pergunta executiva específica — não apenas mostrar dados. RLS por perfil + Bookmarks para navegação contextual entre Liderança, Gestores e Analistas.

% Aderência SLAKPI
% Aderência SLA = DIVIDE( CALCULATE( COUNTROWS(fct_solicitacoes), fct_solicitacoes[fl_dentro_sla] = 1 ), CALCULATE( COUNTROWS(fct_solicitacoes), NOT ISBLANK(fct_solicitacoes[fl_dentro_sla]) ), 0 )
Exclui demandas em aberto do denominador — sem isso, o KPI seria distorcido pelo volume não finalizado.
Lead Time MédioMEASURE
Lead Time Médio = CALCULATE( AVERAGEX( FILTER( fct_solicitacoes, NOT ISBLANK( fct_solicitacoes[nr_lead_time_horas] ) ), fct_solicitacoes[nr_lead_time_horas] ) )
AVERAGEX + FILTER garante que demandas ainda abertas não distorçam a média do tempo de atendimento.
Var SLA MoMTREND
Var SLA MoM = VAR atual = [% Aderência SLA] VAR anterior = CALCULATE( [% Aderência SLA], DATEADD( dim_data[dt_data], -1, MONTH ) ) RETURN DIVIDE(atual - anterior, anterior, 0)
Alimenta o ▲▼ no card de SLA. Usa DATEADD na tabela de datas — mais robusto que cálculo direto na fato.
Taxa RetrabalhoQUALITY
Taxa Retrabalho = DIVIDE( CALCULATE( COUNTROWS(fct_solicitacoes), fct_solicitacoes[fl_correcao] = 1 ), COUNTROWS(fct_solicitacoes), 0 ) -- Threshold de alerta: 5% -- Formatação condicional no card
KPI de qualidade. Formatação condicional muda o card para vermelho quando ultrapassa 5% de retrabalho.
Tempo Ciclo MédioTEMPO
Tempo Ciclo Médio = AVERAGEX( FILTER( fct_solicitacoes, fct_solicitacoes[nm_status] = "Concluída" ), fct_solicitacoes[nr_lead_time_horas] )
Tempo médio do ciclo completo (abertura → entrega) em horas úteis. Exclui demandas ainda abertas para não distorcer a média.
Score PerformanceCOMPOSITE
Score Performance = VAR sla = [% Aderência SLA] * 100 VAR qual = (1 - [Taxa Retrabalho]) * 100 RETURN ROUND( sla * 0.7 + qual * 0.3, 1 ) -- 70% SLA · 30% Qualidade
Score composto para ranking de analistas. Alimenta o visual de tabela com formatação condicional por faixa.

09 · Case · VBA Expert

Ecossistema de
Automação VBA

8 macros interconectadas que automatizaram o ciclo completo de atualização de um documento crítico de precificação de uma grande operadora de telecomunicações. De processos totalmente manuais e repetitivos a um sistema integrado, ágil e com tratamento de erros robusto.

ANTES
Manual · Lento · Frágil

Cada etapa da atualização exigia atenção manual: formatação de cabeçalhos aba por aba, ajuste de cidades, geração de PDF, reordenação de abas. Processo totalmente suscetível a erros humanos.

DEPOIS
Automatizado · Validado · Rastreável

Um clique executa o ciclo completo: lê parâmetros → aplica layout com cor dinâmica → atualiza textos e logos → agrupa cidades por tecnologia/região → ajusta paginação → gera PDF único.

DESIGN
Modularização & Retry Pattern

Cada macro é dividida em sub-rotinas de responsabilidade única. Cópia de imagens usa retry pattern (5–10 tentativas) para garantir robustez em ambientes com latência de clipboard.

''' AtualizarCabecalho — Rotina MESTRE
' Lê a tabela de parâmetros UMA única vez e aplica em sequência:
' 1. Ajuste de Linhas   2. Layout com cor dinâmica
' 3. Textos             4. Logo (Retry Pattern)
Sub AtualizarCabecalho()
    On Error GoTo TrataErro
    Application.ScreenUpdating = False
    Application.Calculation = xlCalculationManual

    For i = 2 To lastRow
        NOMEABA     = Trim(wsParam.Cells(i, 2).Value)
        acaoCriar   = Trim(wsParam.Cells(i, 3).Value)

        If UCase(acaoCriar) = "CRIAR" And AbaExiste(NOMEABA) Then
            Set wsDestino = ThisWorkbook.Worksheets(NOMEABA)
            wsDestino.Activate

            tipoCapitulo = wsParam.Cells(i, 5).Value

            ' 1. Ajustar Linhas
            Call ProcessarLinhas(wsDestino)

            If tipoCapitulo = "SVAS" Or tipoCapitulo = "OPCIONAIS" Then
                ' Capítulos especiais: sem forma colorida
            Else
                ' 2. Cor dinâmica por tipo de capítulo
                corDinamica = ObterCorDoCapitulo(tipoCapitulo)
                ' 3. Formas (layout)
                Call ProcessarFormas(wsDestino, corDinamica)
            End If

            ' 4. Textos e 5. Logo
            Call ProcessarTextos(wsDestino, txtTabela, txtCapitulo, txtPagina)
            Call ProcessarLogo(wsAux, wsDestino, nomeLogo)
        End If
    Next i

    MsgBox "Processo concluído com sucesso!", vbInformation
TrataErro:
    MsgBox "Erro na linha " & i & ": " & Err.Description, vbCritical
End Sub
''' ObterCorDoCapitulo — Cor dinâmica por capítulo
' Retorna a cor RGB com base no tipo de capítulo lido da planilha.
' Azul = Aquisição e Base (padrão de mercado)
' Verde = Rentabilização / Blindagem (diferenciado)
' Âmbar = SVAs e Opcionais (complementar)
Private Const COR_OPCAO1 As Long = 11441949  ' Azul   (#1D97AE)
Private Const COR_OPCAO2 As Long = 1270487   ' Verde  (#50B000)
Private Const COR_OPCAO3 As Long = 192       ' Âmbar  (#FFC000)

Function ObterCorDoCapitulo(tipo As String) As Long
    Select Case UCase(Trim(tipo))
        Case "AQUISIÇÃO E BASE"
            ObterCorDoCapitulo = COR_OPCAO1

        Case "RENTABILIZAÇÃO", "BLINDAGEM"
            ObterCorDoCapitulo = COR_OPCAO2

        Case "SVAS", "OPCIONAIS"
            ObterCorDoCapitulo = COR_OPCAO3

        Case Else
            ObterCorDoCapitulo = COR_OPCAO1  ' Padrão seguro
    End Select
End Function
''' ORDENAR_CANAIS — UDF de normalização e ordenação
' Converte texto para maiúsculas, separa por vírgula,
' remove espaços, aplica Bubble Sort e reune padronizado.
' Uso em célula: =ORDENAR_CANAIS(A1)
Function ORDENAR_CANAIS(Texto As String) As String
    Dim arr() As String
    Dim i As Integer, j As Integer, temp As String

    ' Trava de segurança
    If Trim(Texto) = "" Then
        ORDENAR_CANAIS = ""
        Exit Function
    End If

    Texto = UCase(Texto)
    arr   = Split(Texto, ",")

    ' Kaizen de dados: remove espaços extras
    For i = LBound(arr) To UBound(arr)
        arr(i) = Trim(arr(i))
    Next i

    ' Bubble Sort alfabético
    For i = LBound(arr) To UBound(arr) - 1
        For j = i + 1 To UBound(arr)
            If arr(i) > arr(j) Then
                temp   = arr(i)
                arr(i) = arr(j)
                arr(j) = temp
            End If
        Next j
    Next i

    ' Reune com padrão: vírgula + espaço
    ORDENAR_CANAIS = Join(arr, ", ")
End Function
''' AtualizarFormulas — Leitura e escrita em massa
' Carrega toda a planilha para um array de memória,
' processa lógica de concatenação por categoria (BL, Móvel, TV+...)
' e grava os resultados de volta em uma única operação.
Sub AtualizarFormulas()
    ' Carrega A2:AF_LastRow para memória (a "Grande Otimização")
    vDados = ws.Range("A2:AF" & lastRow).Value

    For i = LBound(vDados, 1) To UBound(vDados, 1)
        valB = Trim(CStr(vDados(i, 2)))  ' Categoria (BL, Móvel, TV+...)
        valH = Trim(CStr(vDados(i, 8)))  ' Tipo (Celular, Fibra...)

        ' Lógica de concatenação por categoria (Switch otimizado)
        Select Case valB
            Case "BANDA LARGA"
                resultadoAF = valC & valE & valH & valI & valJ
            Case "MOVEL"
                If valD = "RENTABILIZAÇÃO" Then
                    resultadoAF = valC & valE & valL & valI & valJ
                Else
                    resultadoAF = valC & valE & valL & valI & valJ & valY
                End If
            Case "CLARO TV+"
                resultadoAF = valC & valE & valI & valH & valJ
        End Select

        ' Soma condicional: se CELULAR, soma colunas Q:U (franquias)
        If UCase(valH) = "CELULAR" Then
            somaI = 0
            For c = 17 To 21  ' Q=17 até U=21
                If IsNumeric(vDados(i, c)) Then somaI = somaI + CDbl(vDados(i, c))
            Next c
            vSaidaI(i, 1) = somaI
        End If
    Next i

    ' Grava tudo de volta em UMA operação (instantâneo)
    ws.Range("AF2:AF" & lastRow).Value = vSaidaAF
    ws.Range("I2:I"  & lastRow).Value = vSaidaI
End Sub
